\frame{\frametitle{Background}
\pause
\begin{itemize}
	\item Cross-stock momentum (CSM): returns of related stock cross-predict each other
	\begin{itemize}
		\item e.g. \citet{cohen2008economic}: ~~customers $\rightarrow$ suppliers
	\end{itemize}

	\vspace{20pt}
	\item Prevailing interpretation: correction of underreaction to cash flow information
	
	\vspace{20pt}
	\item Many upcoming papers with different relatedness measures... 
	\begin{itemize}
		\item stocks w/ overlapping news \hfill {\scriptsize \citep{tao2020news,chen2023news}}
		\item similar characteristics \hfill {\scriptsize \citep{he2021similar}}
		\item correlated bond ratings \hfill {\scriptsize \citep{feng2022economic}}
		\item similar labor force \hfill {\scriptsize \citep{liu2022labor}}
		\item common holdings by institutions \hfill {\scriptsize \citep{ying2024gradual}}... 
%		\item complementary products \hfill {\scriptsize \citep{lee2024production}}...
	\end{itemize}
\end{itemize}
\vspace{30pt}
}

\frame{\frametitle{This paper}
\begin{enumerate}
	\item If CSM \textit{only} reflect correction of underreaction to cash flows, the it should not revert ~\hfill {\footnotesize \citep{samuelson1965proof}}
    \begin{itemize}
        \item However, CSM returns \textit{revert}, suggesting it is more complex
    \end{itemize}

	\vspace{15pt}
	\item Decompose CSM into two parts
	\begin{itemize}
		\item \textcolor{red}{Unidirectional lead-lag} : pure underreaction
		\item \textcolor{olive}{(Generalized) factor momentum}: also contains overreaction
	\end{itemize}
	\vspace{20pt}
\end{enumerate}

$\Rightarrow$ we propose future studies conduct similar decompositions to shed light on the economic interpretation of CSM. 

\vspace{20pt}
}

\begin{frame}
    \frametitle{What about long-horizon returns?}

    \begin{itemize}
        \item If CSM returns reflect underreaction to cash flow news, then the returns should not revert ~\hfill {\scriptsize \citep{samuelson1965proof}}
    
        \vspace{10pt}
        \item For presentation, we use a ``combined'' CSM predictor
        \begin{align*}
            s_{i,t} & = \frac{1}{K_{i,t}} \sum_{k = 1}^{K_{i,t}} s_{i,t}^k ~~~ \forall \text{ stock }i
        \end{align*}

        \item To examine long horizon returns, for each vintage $t$ and each horizon $h$, compute:
        \begin{align*}
            R_{t+h | t} & \equiv s_t' ~r_{t+h} & \text{(one-period port return)} \\
            R_{t+1 \rightarrow t+h| t} & \equiv \prod_{s=1}^h R_{t+s|t} & \text{(cumulative port return)}
        \end{align*}
    \end{itemize}
\end{frame}

\begin{frame}[label=Pi]
\frametitle{CSM construction isn't \textit{tailored} for underreaction}
\begin{itemize}
\item Underreaction: information is reflected in some stocks faster
\begin{itemize}
	\item Suggests \textit{uni-directional} lead-lag: \hfill $\underbrace{\text{lead stock }}_\text{faster} \rightarrow \underbrace{\text{lag stock}}_\text{slower}$
\end{itemize}
\vspace{10pt}
\item However, typical CSM construction also captures other effects
\begin{itemize}
% \item For each stock $i$, CSM predictor is $s_{i,t} = \sum_{j \in \textit{lead}(i)} \Pi_{i,j} \cdot r_{j,t}$
% \vspace{5pt}
\item \textbf{Notation:} each CSM predictor is characterized by a different ``prediction matrix'' $\textcolor{blue}{\Pi}$: \hfill {\scriptsize ~\citep{kelly2023principal}}



\begin{align*}
E\begin{bmatrix} r_{1,t+1} \\ r_{2,t+1} \\ \vdots \\ r_{N,t+1} \end{bmatrix}  & = \textcolor{blue}{\begin{bmatrix} \Pi_{1,1} & \Pi_{1,2} & \ldots & \Pi_{1,N} \\
\Pi_{2,1} & \Pi_{2,2} & \ldots & \Pi_{2,N} \\
\vdots & & \ddots & \vdots \\
\Pi_{N,1} & \Pi_{N,2} & \ldots & \Pi_{N,N} \\
 \end{bmatrix}} \cdot \begin{bmatrix} r_{1,t} \\ r_{2,t} \\ \vdots \\ r_{N,t} \end{bmatrix}
\end{align*}
\item CSM constructions include both directions ($i \rightarrow j$ and $j \rightarrow i$)
\end{itemize}
\end{itemize}
\vspace{40pt}
\end{frame}

\begin{frame}[label=Pi_industry]
\frametitle{Example: industry momentum}
\begin{itemize}
\item Consider four stocks with industry groupings $\{\{1,2\},\{3,4\}\}$, and market cap = \$1, \$2, \$3, \$4
\vspace{10pt}
\item Then, the associated prediction matrix is:

\begin{align*}
E\begin{bmatrix} r_{1,t+1} \\ r_{2,t+1} \\ r_{3,t+1} \\ r_{4,t+1} \end{bmatrix} & = \underbrace{\begin{bmatrix} 1/3 & 2/3 & \cdot & \cdot \\
1/3 & 2/3 & \cdot & \cdot \\ 
\cdot & \cdot & 3/7& 4/7 \\
\cdot & \cdot & 3/7& 4/7
\end{bmatrix}}_{=\Pi} \cdot \begin{bmatrix} r_{1,t} \\ r_{2,t} \\ r_{3,t} \\ r_{4,t} \end{bmatrix}
\end{align*}
\begin{itemize}
\item Other examples: \hyperlink{Pi_tech}{\beamergotobutton{technology link}} ~ \hyperlink{Pi_econ}{\beamergotobutton{economic link}}
\end{itemize}

\vspace{5pt}
\item What effects does this $\Pi$ capture?
\begin{itemize}
    \item Some within industry lead-lag: 2 $\rightarrow$ 1, 4 $\rightarrow$ 3
    \item But also momentum in industry returns \\
    \textcolor{blue}{$\Rightarrow$ Will be good to separate the two!}
\end{itemize}
\end{itemize}
\end{frame}

\begin{frame}[label=decomposition]
\frametitle{\citet{kelly2023principal} decomposition}
\begin{itemize}
	\item Decompose prediction matrix:
	\begin{align*}
    \Pi &= \underbrace{\left(\frac{\Pi+\Pi'}{2}\right)}_{\equiv \Pi_\text{sym}} + \underbrace{\left(\frac{\Pi-\Pi'}{2}\right)}_{\equiv \Pi_\text{asy}}
    \end{align*}
	
    \item Properties:
    \begin{itemize}
        \item \textbf{Symmetric part}~($\Pi_\text{sym} = \Pi_\text{sym}'$): (generalized) factor momentum \\ ~\hfill \hyperlink{understanding_sym}{\beamergotobutton{details}}
        \vspace{20pt}
 %    \begin{align*}
	% 	\Pi_\text{sym} & = W \cdot \Lambda \cdot W' &  \text{(PCA)}\\
 %        Cov(f_{t+1}, f_t') & = \underbrace{\begin{bmatrix}
 %        \lambda_1 & 0 & 0\ldots & 0 \\ 
 %        0 & \lambda_2 & 0 \ldots & 0 \\
 %        0 & 0 & 0 \ldots & \lambda_N
 %        \end{bmatrix}}_{= \Lambda} & \text{ where }
	% 	f_t \equiv W' \cdot r_t
	% \end{align*}     

    \item \textbf{Asymmetic part} ~($\Pi_\text{asy} = -\Pi_\text{asy}'$): unidirectional lead-lags
        \begin{itemize}
        \item $\Pi_{\text{asy},i,j} = - \Pi_{\text{asy},j,i}$: either $i$ leads $j$ or the reverse
        \end{itemize}
    \end{itemize} 
\end{itemize}
\vspace{20pt}
\end{frame}

\begin{frame}
\frametitle{We decompose CSM portfolio returns}

\begin{itemize}
    \item Decomposition:
    \begin{align*}
        s_t & = \Pi ~r_t = \underbrace{\Pi_\text{sym} ~r_t}_{=s_{\text{sym},t}} + \underbrace{\Pi_\text{asy} ~r_t}_{=s_{\text{asy},t}} & \text{(signal)}\\
        R_{t+h | t} & = s_t' ~r_{t+h} = s_{\text{sym},t}' ~r_{t+h} + s_{\text{asy},t}' ~r_{t+h} &\text{(portfolio return)}
    \end{align*}

    \vspace{10pt}
    \item We conjecture:
    \begin{itemize}
        \item Asymmetric component: captures underreaction
        \vspace{10pt}
        \item Symmetric component: captures reversals

\begin{frame}[label=understanding_sym]
    \frametitle{Understanding the symmetric component}

\begin{itemize}
    \item Intends to capture predictive relationship:
    \begin{align*}
        E_t(r_{t+1}) & = \Pi_\text{sym} ~ r_t
    \end{align*}
    \begin{itemize}
        \item Effectively, predicting that $Cov(r_{t+1}, r_t') = \Pi_\text{sym}$
    \end{itemize}

    \vspace{10pt}
    \item We can rotate from stock-space to factor-space via PCA:
    \begin{align*}
        \Pi_\text{sym} & = W \cdot \Lambda \cdot W' \\
        \text{ where } f_t & = W'~r_t, \text{ and } \Lambda \text{ is diagonal}
    \end{align*} 

    \vspace{10pt}
    \item Then, this amounts to predicting that:
    \begin{align*}
        Cov(f_{t+1}, f_t') & = \Lambda
    \end{align*}
\end{itemize}

\begin{frame}[label=fm_as_sym_explanation]
    \frametitle{Factor momentum $\approx$ CSM with symmetric $\Pi$}

\begin{itemize}
    \item To wit, (TS)FM in {\scriptsize \citet{gupta2019factor}, \citet{arnott2023factor}}:
\begin{itemize}
    \item Start from $K$ known factor portfolios (e.g. size) with portfolio weights $\{w_k\}_{k=1}^K$. Factor returns are $f_{k,t} = w_k' r_{t}$ ~~~~~ ($f_t = W' r_t$)

    \item Bet on factor return continuation:
\end{itemize}
\end{itemize}
\begin{align*}
    E_t\begin{bmatrix}f_{1,t+1} \\ \vdots \\ f_{K,t+1} \end{bmatrix} & = \begin{bmatrix} \lambda & \ldots & \cdot \\
    \cdot & \ddots & \cdot \\
    \cdot & \ldots & \lambda
    \end{bmatrix} ~ \begin{bmatrix}f_{1,t} \\ \vdots \\ f_{K,t} \end{bmatrix} \\
    \Leftrightarrow E_t(W' r_{t+1}) & = \Lambda W' r_t \\
    \Rightarrow E_t(r_{t+1}) & \approx \underbrace{(W')^{+} \Lambda W'}_\text{Low-Rank Symmetric} r_t
\end{align*}
\footnotesize{Where $(W')^{+} = W(W'W)^{-1}$ is the Moore-Penrose pseudoinverse.}\hfill \hyperlink{sym_as_fm}{\beamergotobutton{back}}
\end{frame}
